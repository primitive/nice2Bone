	function primitive_tag( $object, $field_name, $request ) {
	$formatted_tags = array();
	$tags = get_tags( $object['id'] );

    foreach ($tags as $tag) {
		$formatted_tags = $tag->name;
    }

    return $formatted_tags;
}
